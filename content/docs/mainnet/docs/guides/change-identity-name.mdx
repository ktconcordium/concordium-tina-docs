title: "Change identity name"

(change-mw-id-name)=


<accordion
  heading="Concordium Wallet"
  docText={<>
1. Open *Wallet Settings* by tapping the gear icon in the lower right of the main screen.

2. Tap **Identities**.

3. Tap the identity you want to edit.

4. On the identity card, tap the edit icon next to the identity name.

5. Enter the new identity name and tap **Save**.
  </>}
  image=""
  fullWidth={true}
/>

<accordion
  heading="Concordium Wallet for Web"
  docText={<>
1. Open the menu and select ID Cards. If you have more than one identity, select the ID card you want to edit.

2. Click **Edit Name** next to the identity name.

3. Change the name and click **Save**.
  </>}
  image=""
  fullWidth={true}
/>
